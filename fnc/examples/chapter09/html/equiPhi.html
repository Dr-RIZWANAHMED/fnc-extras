We plot over the interval with equispaced nodes for different values of n.
x = linspace(-1,1,1601)';
Phi = zeros(size(x));
for n = 10:10:50
t = linspace(-1,1,n+1)';
for k = 1:length(x)
Phi(k) = prod(x(k)-t);
end
semilogy(x,abs(Phi)), hold on
end
axis tight, title('Effect of equispaced nodes') % ignore this line
xlabel('x'), ylabel('|\Phi(x)|') % ignore this line
