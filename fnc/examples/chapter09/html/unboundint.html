f = @(x) 1./(1+x.^2);
h = logspace(-3,-1,60);
M = logspace(3,16,60);
for i = 1:length(h)
for j = 1:length(M)
I(i,j) = intde(f,h(i),M(j));
end
end
err = abs(I-pi);
contourf(h,M,-log10(err)')
set(gca,'xsc','log','xdir','rev','ysc','log') % ignore this line
colorbar % ignore this line
xlabel('h'), ylabel('M') % ignore this line
title('Number of accurate digits') % ignore this line
